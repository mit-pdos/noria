VIEW BOUNDARY_notifications: SELECT BOUNDARY_replying_comments_for_count.user_id, COUNT(*) AS notifications FROM `BOUNDARY_replying_comments_for_count` GROUP BY `BOUNDARY_replying_comments_for_count`.`user_id`;
